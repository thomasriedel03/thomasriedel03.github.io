1.3.2 repairTheStreet 


            

                Lösung:
            

            void repairTheStreet(){
                repeat(9){
                    if(rightIsClear()){
                        turnRight();
                        moveForward();
                        dropBeeper();
                        turnAround();
                        moveForward();
                        turnRight();
                        moveForward();
                        
                    }
                    
                    else{ 
                        moveForward();
                    }
                    
                }  
                turnRight();
                moveForward();
                dropBeeper();
                turnAround();
            }

1.3.3 cleanTheRoom 

            

                Lösung:
            

            void cleanTheRoom(){
                while(frontIsClear()){
                    cleanRow();  
                    leftCircle();
                    cleanRow();
                    rightCircle();  
                }  
            }   
            
            void cleanRow(){
                while(frontIsClear()||onBeeper()){
                    if(onBeeper()){
                        pickBeeper();
                    }
                    else{
                        moveForward();
                    }     
                }    
            }
            
            void leftCircle(){
                turnLeft();
                moveForward();
                turnLeft();
            }
            
            void rightCircle(){
                turnRight();
                moveForward();
                turnRight();
            }
